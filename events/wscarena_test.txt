        }
    }

planet_event = {                                        # ascension ship ascends
    id = wscarena_test.57
    is_triggered_only = yes
    hide_window = yes

    trigger = {
        solar_system = {
            any_ship_in_system = {
                distance = { source = root min_distance >= 0 max_distance <= 30 type = euclidean }
                is_ship_size = ascension_ship
            }    
        } 
    }

    immediate = {
        while = {
            count = 5
            solar_system = {
                    random_fleet_in_system = {
                        limit = {
                            distance = { source = root min_distance >= 0 max_distance <= 30 type = euclidean }
                            any_owned_ship = {
                                is_ship_size = ascension_ship
                            }
                        }
                            random_owned_ship = {
                                limit = {
                                    is_ship_size = ascension_ship
                                }
                                destroy_ship = this
                                fleet = {
                                    owner = {
                                        change_variable = {
                                            which = ascended_pops
                                            value = 1
                                        }
                                    }
                                }
                            }
                        }
                    }
                }

    }
